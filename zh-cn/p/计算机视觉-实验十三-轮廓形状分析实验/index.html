		minEnclosingCircle( contours_poly[i], center[i], radius[i] );//å¯¹ç»™å®šçš„ 2Dç‚¹é›†ï¼Œå¯»æ‰¾æœ€å°é¢ç§¯çš„åŒ…å›´åœ†å½¢

	//Mat drawing = Mat::zeros( threshold_output.size(), CV_8UC3 );
