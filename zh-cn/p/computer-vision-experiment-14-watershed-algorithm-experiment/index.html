Mat g_maskImage, g_srcImage;

//-----------------------------------„ÄêÂÖ®Â±ÄÂáΩÊï∞Â£∞ÊòéÈÉ®ÂàÜ„Äë--------------------------------------
//		ÊèèËø∞ÔºöÂÖ®Â±ÄÂáΩÊï∞ÁöÑÂ£∞Êòé
//-----------------------------------------------------------------------------------------------
static void ShowHelpText();
static void on_Mouse( int event, int x, int y, int flags, void* );


//-----------------------------------„Äêmain( )ÂáΩÊï∞„Äë--------------------------------------------
//		ÊèèËø∞ÔºöÊéßÂà∂Âè∞Â∫îÁî®Á®ãÂ∫èÁöÑÂÖ•Âè£ÂáΩÊï∞ÔºåÊàë‰ª¨ÁöÑÁ®ãÂ∫è‰ªéËøôÈáåÂºÄÂßãÊâßË°å
//-----------------------------------------------------------------------------------------------
int main( int argc, char** argv )
{	
	//„Äê0„ÄëÊîπÂèòconsoleÂ≠ó‰ΩìÈ¢úËâ≤
	system("color 6F"); 

	//„Äê0„ÄëÊòæÁ§∫Â∏ÆÂä©ÊñáÂ≠ó
	ShowHelpText( );

	//„Äê1„ÄëËΩΩÂÖ•ÂéüÂõæÂπ∂ÊòæÁ§∫ÔºåÂàùÂßãÂåñÊé©ËÜúÂíåÁÅ∞Â∫¶Âõæ
	g_srcImage = imread("1.jpg", 1);
	imshow( WINDOW_NAME1, g_srcImage );
	Mat srcImage,grayImage;
	g_srcImage.copyTo(srcImage);
	cvtColor(g_srcImage, g_maskImage, COLOR_BGR2GRAY);
	cvtColor(g_maskImage, grayImage, COLOR_GRAY2BGR);
	g_maskImage = Scalar::all(0);

	//„Äê2„ÄëËÆæÁΩÆÈº†Ê†áÂõûË∞ÉÂáΩÊï∞
	setMouseCallback( WINDOW_NAME1, on_Mouse, 0 );

	//„Äê3„ÄëËΩÆËØ¢ÊåâÈîÆÔºåËøõË°åÂ§ÑÁêÜ
	while(1)
	{
		//Ëé∑ÂèñÈîÆÂÄº
		int c = waitKey(0);

		//Ëã•ÊåâÈîÆÈîÆÂÄº‰∏∫ESCÊó∂ÔºåÈÄÄÂá∫
		if( (char)c == 27 )

		//ÊåâÈîÆÈîÆÂÄº‰∏∫2Êó∂ÔºåÊÅ¢Â§çÊ∫êÂõæ
		if( (char)c == '2' )
		{

			srcImage.copyTo(g_srcImage);
			imshow( "image", g_srcImage );
		}

		//Ëã•Ê£ÄÊµãÂà∞ÊåâÈîÆÂÄº‰∏∫1ÊàñËÄÖÁ©∫Ê†ºÔºåÂàôËøõË°åÂ§ÑÁêÜ
		if( (char)c == '1' || (char)c == ' ' )
		{
			//ÂÆö‰πâ‰∏Ä‰∫õÂèÇÊï∞
			int i, j, compCount = 0;
			vector<vector<Point> > contours;
			vector<Vec4i> hierarchy;

			//ÂØªÊâæËΩÆÂªì
			findContours(g_maskImage, contours, hierarchy, RETR_CCOMP, CHAIN_APPROX_SIMPLE);

			//ËΩÆÂªì‰∏∫Á©∫Êó∂ÁöÑÂ§ÑÁêÜ
			if( contours.empty() )
				continue;

			//Êã∑Ë¥ùÊé©ËÜú
			Mat maskImage(g_maskImage.size(), CV_32S);
			maskImage = Scalar::all(0);

			//Âæ™ÁéØÁªòÂà∂Âá∫ËΩÆÂªì
			for( int index = 0; index >= 0; index = hierarchy[index][0], compCount++ )
				drawContours(maskImage, contours, index, Scalar::all(compCount+1), -1, 8, hierarchy, INT_MAX);

			//compCount‰∏∫Èõ∂Êó∂ÁöÑÂ§ÑÁêÜ
			if( compCount == 0 )
				continue;

			//ÁîüÊàêÈöèÊú∫È¢úËâ≤
			vector<Vec3b> colorTab;
			for( i = 0; i < compCount; i++ )
			{
				int b = theRNG().uniform(0, 255);
				int g = theRNG().uniform(0, 255);
				int r = theRNG().uniform(0, 255);

				colorTab.push_back(Vec3b((uchar)b, (uchar)g, (uchar)r));
			}

			//ËÆ°ÁÆóÂ§ÑÁêÜÊó∂Èó¥Âπ∂ËæìÂá∫Âà∞Á™óÂè£‰∏≠

			Mat watershedImage(maskImage.size(), CV_8UC3);
			for( i = 0; i < maskImage.rows; i++ )
				for( j = 0; j < maskImage.cols; j++ )
				{
					int index = maskImage.at<int>(i,j);
					if( index == -1 )

	}

//-----------------------------------„ÄêonMouse( )ÂáΩÊï∞„Äë---------------------------------------
//		ÊèèËø∞ÔºöÈº†Ê†áÊ∂àÊÅØÂõûË∞ÉÂáΩÊï∞
//-----------------------------------------------------------------------------------------------
static void on_Mouse( int event, int x, int y, int flags, void* )
{
	//Â§ÑÁêÜÈº†Ê†á‰∏çÂú®Á™óÂè£‰∏≠ÁöÑÊÉÖÂÜµ
	if( x < 0 || x >= g_srcImage.cols || y < 0 || y >= g_srcImage.rows )
		return;

	//Â§ÑÁêÜÈº†Ê†áÂ∑¶ÈîÆÁõ∏ÂÖ≥Ê∂àÊÅØ
	if( event == EVENT_LBUTTONUP || !(flags & EVENT_FLAG_LBUTTON) )
		prevPt = Point(-1,-1);
	else if( event == EVENT_LBUTTONDOWN )
		prevPt = Point(x,y);

	//Èº†Ê†áÂ∑¶ÈîÆÊåâ‰∏ãÂπ∂ÁßªÂä®ÔºåÁªòÂà∂Âá∫ÁôΩËâ≤Á∫øÊù°
	else if( event == EVENT_MOUSEMOVE && (flags & EVENT_FLAG_LBUTTON) )
	{
		Point pt(x, y);
		if( prevPt.x < 0 )
			prevPt = pt;
		line( g_maskImage, prevPt, pt, Scalar::all(255), 5, 8, 0 );
		line( g_srcImage, prevPt, pt, Scalar::all(255), 5, 8, 0 );
		prevPt = pt;
		imshow(WINDOW_NAME1, g_srcImage);
	}
