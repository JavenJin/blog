#include<stdio.h>
int main()
{
	int M,m,a[10000],b[10000],c[10000],sum_a,sum_b,j;
	scanf("%d",&M); 
	m = M;
	while(m)
	{
		scanf("%d%d",&a[m],&b[m]);
		m--;
	}
	for(m=1;m<=M;m++)
	{
		sum_a = 0;
		sum_b = 0;
		for(j=1;j<a[m];j++)
		{
			if(a[m]%j == 0)
				sum_a+=j;
		}
		for(j=1;j<b[m];j++)
		{
			if(b[m]%j == 0)
				sum_b+=j;
		}
//		printf("a=%d\tb=%d\n",a[m],b[m]);
//		printf("suma=%d\tsumb=%d",sum_a,sum_b);
		if((sum_a==b[m])&&(sum_b==a[m]))
			c[m]=1;
		else
			c[m]=0;
//		printf("\n%d",c[m]);
	}
	for(j=M;j>=1;j--)
	{
		if(c[j]==1)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
