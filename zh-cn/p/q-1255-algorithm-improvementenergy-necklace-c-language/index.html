int main()
{
	long long int dp[202][202];
	long long int s[202];
	memset(dp,0,sizeof(dp));
	long long int sum=0;
	int n,i,j,k,len;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
        scanf("%lld",&s[i]),s[i+n]=s[i];
    for(len=2;len<=n;len++)
        for(i=1;i+len-1<2*n;i++)
        {
            j=i+len-1;
      	    for(k=i;k<j;k++)
      	        dp[i][j]=Max(dp[i][j],dp[i][k]+dp[k+1][j]+s[i]*s[k+1]*s[j+1]);

	    if(sum<dp[i][i+n-1])sum=dp[i][i+n-1];
