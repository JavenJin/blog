void dijkstra_syjd()
{
	int v0;
	printf("Please enter the Attraction of your query:");
	scanf("%d",&v0);
	boolean final[MaxViewNum];
	int i,k,j,v,min,x;
	for(v=1;v<=N;v++)
	{
	final[v]=FALSE;
	d[v]=M.length[v0][v];
	if(d[v]<MaxRoad&&d[v]!=0)
	p[v]=v0;
	else
	p[v]=-1;
	}
	final[v0]=TRUE;
	d[v0]=0;
	for(i=2;i<=N;i++)
	{
		min=MaxRoad;
		for(k=1;k<=N;++k)
		if(!final[k]&&d[k]<min)
		{
			v=k;
			min=d[k];
		}
	
		if(min==MaxRoad)
		return;
		final[v]=TRUE;
		for(k=1;k<=N;++k)
		if(!final[k]&&(min+M.length[v][k]<d[k]))
		{
			d[k]=min+M.length[v][k];
			p[k]=v;
		}
	}
}

void printsyjd()
{
	int st[MaxViewNum],i,pre,top=-1;
	for(i=1;i<=N;i++)
	{
		printf("\nArrival Attraction %2d The total distance is %5d , The path passed through:",i,d[i]);
		st[++top]=i;
		pre=p[i];
		while(pre!=-1)
		{
			st[++top]=pre;
			pre=p[pre];
		}
	while(top>0)
	{
	printf("%d",st[top--]);
	if(top>0)
	printf("---");
	}
	}
	getch();
}
